# 순열 작성해보기
## 사전식으로 나오게됨. 최소변경을 통한 방법도 있는데 나는 이방식을 사용하고있음... 
``` java
import java.util.Arrays;

public class Permutation {
	static int[] arr;
	static int[] rst;
	static boolean[] v;
	static int n;
	public static void main(String[] args) {
		arr = new int[] { 1, 2, 3};
		n = arr.length;
		rst = new int[n];
		v = new boolean[n];
		perm(0);
	}
	static void perm(int cnt) {
		if(cnt == n) {
			System.out.println(Arrays.toString(rst));
			return;
		}
		for(int i = 0; i < n; i++) {
			if(v[i]) {
				continue;
			}
			v[i] = true;
			rst[cnt] = arr[i];
			perm(cnt+1);
			v[i] = false;
		}
	}
}
```
